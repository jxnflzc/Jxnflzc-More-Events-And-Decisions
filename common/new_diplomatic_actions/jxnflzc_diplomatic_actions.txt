jm_annex = {
	category = relation
	
    require_acceptance = no
    
    is_visible = {
		not = {
			ai = yes
		}
		has_country_flag = jm_diplomatic_on
    }
    
    is_allowed = {
		not = {
			ai = yes
		}
		has_country_flag = jm_diplomatic_on
		FROM = {
			vassal_of = ROOT
		}
    }
    
    on_accept = {
		FROM = {
			every_owned_province = {
				cede_province = ROOT
				if = {
					limit = {
						not = {
							is_core = ROOT
						}
					}
					add_core = ROOT
				}
			}
		}
    }
    
    ai_will_do = {
        always = no
    }
}
